# Lua packages configuration for install_lua_deps.
# Lines starting with # are comments.
#
# Package line: pkg <rock_name> [version]
# Install target line: install <os> <target_ref>
# Install arg line: arg <os> <single_arg>
# Env line: env <os> <name> <value_ref>
# Dependency line: dep <name> <os> <method> <url_or_hint>
# Dependency variable line: depvar <dep_name> <os> <var_name> <value_ref>
#
# OS values: windows, linux, macos, any
# Method values: none (pure lua), vcpkg, pkg-config, bundled, cmake
# target_ref / value_ref supports:
#   dep:<name>[/subpath]
#   tool:<subpath>
#   path:<project-relative-path>
#   literal text (returned as-is)
#
# ---------------------------------------------------------------
